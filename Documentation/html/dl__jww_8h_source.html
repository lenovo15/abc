293  static double toReal(const char* value, double def=0.0) {
294  if (value!=NULL && value[0] != '\0') {
295  double ret;
296  if (strchr(value, ',') != NULL) {
297  char* tmp = new char[strlen(value)+1];
298  strcpy(tmp, value);
299  DL_WriterA::strReplace(tmp, ',', '.');
300  ret = atof(tmp);
301  delete[] tmp;
302  }
303  else {
304  ret = atof(value);
305  }
306  return ret;
307  } else {
308  return def;
309  }
310  }
