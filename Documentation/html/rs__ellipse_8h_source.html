330 class EllipseDistanceFunctor
331 {
332 public:
333  EllipseDistanceFunctor(RS_Ellipse* ellipse, double const& target) : distance(target)
334  { // Constructor
335  e=ellipse;
336  ra=e->getMajorRadius();
337  k2=1.- e->getRatio()*e->getRatio();
338  }
339  void setDistance(const double& target){
340  distance=target;
341  }
342 #if BOOST_VERSION > 104500
343  boost::math::tuple<double, double, double> operator()(double const& z) const {
344 #else
345  boost::fusion::tuple<double, double, double> operator()(double const& z) const {
346 #endif
347  double cz=cos(z);
348  double sz=sin(z);
349  //delta amplitude
350  double d=sqrt(1-k2*sz*sz);
351  // return f(x), f'(x) and f''(x)
352 #if BOOST_VERSION > 104500
353  return boost::math::make_tuple(
354 #else
355  return boost::fusion::make_tuple(
356 #endif
357  e->getEllipseLength(z)-distance,
358  ra*d,
359  k2*ra*sz*cz/d
360  );
361  }
362 
363 private:
364 
365  double distance;
366  RS_Ellipse* e;
367  double ra;
368  double k2;
369 };
